Let’s play!

  • animation-duration

    /* s | ms [if no unit: s assumed] */
    
  #square {
    animation-duration: 0s;
  }
            
    -webkit-animation: slide 1s;

  • animation-delay

    /* s | ms [if no unit: s assumed] */
    
  #square {
    animation-delay: 0s;
  }
            
    -webkit-animation: slide 1s 1s;

  • animation-iteration-count

    /* number | infinite */
    
  #square {
    animation-iteration-count: 1;
  }
            
    -webkit-animation: slide 1s 3;

  • animation-direction

    /* normal | alternate | reverse | alternate-reverse */
    
  #square {
    animation-direction: normal;
  }
            
    -webkit-animation: slide 1s 2 alternate;

  • animation-fill-mode

    /* none | forwards | backwards | both */
    
  #square {
    animation-fill-mode: none;
  }
            
    -webkit-animation: slide 1s 1s forwards;

  • animation-timing-function

    /*
cubic-bezier: x1, y1, x2, y2
  [linear, ease, ease-in, ease-in-out, ease-out]

steps: number_of_steps, direction
  [step-start, step-end]
*/
    
  #square {
    animation-timing-function: ease;
  }
            
    -webkit-animation: slide 1s linear;

  • animation-play-state

    /*
running | paused
Note: Can’t use w/ shorthand property `animation:`
*/
    
  #square {
    animation-play-state: running;
  }

Cube rotation


  scene {
    -webkit-perspective: 1000px;
    -webkit-animation: scene 2s alternate infinite both;
  }
  cube { -webkit-animation: cube 2s infinite linear; }

  @-webkit-keyframes cube {
    from { -webkit-transform: rotateY(0) rotateX(0); }
    to   { -webkit-transform: rotateY(360deg) rotateX(360deg); }
  }
  
  @-webkit-keyframes scene {
    from { -webkit-perspective: 2000px; }
    to   { -webkit-perspective: 300px; }
  }
